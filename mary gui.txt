-- Script de Teletransporte
local Players = game:GetService("Players")

function teleportPlayer(playerName, targetPlayerName)
    local player = Players:FindFirstChild(playerName)
    local targetPlayer = Players:FindFirstChild(targetPlayerName)
    
    if player and targetPlayer then
        player.Character:SetPrimaryPartCFrame(targetPlayer.Character.PrimaryPart.CFrame)
    else
        print("Jogador não encontrado")
    end
end

-- Script de ESP
local function createESP(player, color)
    local highlight = Instance.new("Highlight")
    highlight.Parent = player.Character
    highlight.Adornee = player.Character
    highlight.FillColor = color
    highlight.FillTransparency = 0.5
    highlight.OutlineColor = color
    highlight.OutlineTransparency = 0
end

Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(character)
        if player.Team == game.Teams.Allies then
            createESP(player, Color3.fromRGB(0, 0, 255)) -- Azul para aliados
        else
            createESP(player, Color3.fromRGB(255, 0, 0)) -- Vermelho para inimigos
        end
    end)
end)

-- Script de Invisibilidade
local function toggleInvisibility(player, state)
    if player.Character then
        for _, part in pairs(player.Character:GetChildren()) do
            if part:IsA("BasePart") then
                part.Transparency = state and 1 or 0
                part.CanCollide = not state
            end
        end
    end
end

-- Script de Imortalidade (God Mode)
local function toggleGodMode(player, state)
    if player.Character then
        for _, part in pairs(player.Character:GetChildren()) do
            if part:IsA("Humanoid") then
                part.MaxHealth = state and math.huge or 100
                part.Health = part.MaxHealth
            end
        end
    end
end

-- Script de TP Tool
