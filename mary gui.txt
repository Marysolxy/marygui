    button.BorderColor3 = Color3.fromRGB(64, 64, 64) -- Borda cinza escuro
    button.TextColor3 = Color3.fromRGB(255, 255, 255) -- Letras brancas
    return button
end

-- Script de Teletransporte
local Players = game:GetService("Players")

function teleportPlayer(playerName, targetPlayerName)
    local player = Players:FindFirstChild(playerName)
    local targetPlayer = Players:FindFirstChild(targetPlayerName)
    
    if player and targetPlayer then
        player.Character:SetPrimaryPartCFrame(targetPlayer.Character.PrimaryPart.CFrame)
    else
        print("Jogador não encontrado")
    end
end

-- Script de ESP
function activateESP()
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= game.Players.LocalPlayer then
            local highlight = Instance.new("Highlight")
            highlight.Parent = player.Character
            highlight.FillColor = player.TeamColor.Color
            highlight.OutlineColor = Color3.new(1, 1, 1)
        end
    end
end

-- Script de Invisibilidade
local isInvisible = false

function toggleInvisibility()
    local player = game.Players.LocalPlayer
    if isInvisible then
        player.Character.HumanoidRootPart.Transparency = 0
        player.Character.Head.Transparency = 0
        for _, part in pairs(player.Character:GetChildren()) do
            if part:IsA("BasePart") then
                part.Transparency = 0
            end
        end
        isInvisible = false
    else
        player.Character.HumanoidRootPart.Transparency = 1
        player.Character.Head.Transparency = 1
        for _, part in pairs(player.Character:GetChildren()) do
            if part:IsA("BasePart") then
                part.Transparency = 1
            end
        end
        isInvisible = true
    end
end

-- Script de God Mode
local isGodMode = false

function toggleGodMode()
    local player = game.Players.LocalPlayer
    if isGodMode then
        player.Character.Humanoid.MaxHealth = 100
        player.Character.Humanoid.Health = 100
        isGodMode = false
    else
        player.Character.Humanoid.MaxHealth = math.huge
        player.Character.Humanoid.Health = math.huge
        isGodMode = true
    end
end
